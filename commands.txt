## Installing the aws_s3 extension
psql -h parallel-db-um.cvm4cgeo8imx.us-east-1.rds.amazonaws.com -p 5432 -U postgres -W

## Link to install aws extension
--# https://docs.aws.amazon.com/AmazonRDS/latest/UserGuide/USER_PostgreSQL.S3Import.InstallExtension.html
CREATE EXTENSION aws_s3 CASCADE;

## Attach policy to role
aws iam attach-role-policy \
    --policy-arn arn:aws:iam::767398088866:policy/postgresS3Policy-yml-4 \
    --role-name postgresql-S3-Role-yml-4

## Attach role to database
aws rds add-role-to-db-instance \
   --db-instance-identifier parallel-db-um \
   --feature-name s3Import \
   --role-arn arn:aws:iam::767398088866:role/postgresql-S3-Role-yml-4 \
   --region us-east-1
